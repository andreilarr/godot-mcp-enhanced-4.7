```typescript
// src/core/response-limiter.ts
/**
 * Response truncation — Phase 3b
 *
 * Dual-threshold response size control:
 * - 2MB (soft): add warning content block
 * - 4MB (hard): trim largest array, add truncation notice
 *
 * Feature flag: GODOT_MCP_RESPONSE_LIMIT (default: true)
 */

import type { ToolResult } from '../types.js';
import { isFeatureEnabled } from './feature-flags.js';

const SOFT_LIMIT = 2 * 1024 * 1024;  // 2MB
const HARD_LIMIT = 4 * 1024 * 1024;  // 4MB

/**
 * Apply response size limits. Returns same reference (no copy) if under soft limit.
 * Returns new ToolResult with warning/truncation content blocks if limits exceeded.
 */
export function truncateResponse(response: ToolResult): ToolResult {
  if (!isFeatureEnabled('RESPONSE_LIMIT')) return response;

  const text0 = response.content?.[0];
  if (!text0 || text0.type !== 'text') return response;

  const size = text0.text.length;

  // Under soft limit — pass through
  if (size <= SOFT_LIMIT) return response;

  // Between soft and hard — add warning only
  if (size <= HARD_LIMIT) {
    return {
      ...response,
      content: [
        ...response.content,
        { type: 'text' as const, text: '[Warning: Response exceeds 2MB recommended size. Consider using page_size for targeted results.]' },
      ],
    };
  }

  // Over hard limit — truncate
  let parsed: unknown;
  try { parsed = JSON.parse(text0.text); } catch { return response; }

  const trimmed = trimToArrayLimit(parsed, HARD_LIMIT - 200); // reserve 200 bytes for wrapper
  return {
    ...response,
    content: [
      { type: 'text' as const, text: JSON.stringify({ ...trimmed, partial: true, originalSize: size }) },
      { type: 'text' as const, text: '[Response truncated at 4MB. Use page_size parameter for targeted results.]' },
    ],
  };
}

/**
 * Trim the largest array in a data structure to fit within byte limit.
 * Preserves non-array fields and adds truncatedAt + totalNodeCount metadata.
 */
export function trimToArrayLimit(data: unknown, limitBytes: number): unknown {
  if (typeof data !== 'object' || data === null) return data;
  const obj = data as Record<string, unknown>;

  // Find the largest array
  let largestKey = '';
  let largestLen = 0;
  for (const [key, val] of Object.entries(obj)) {
    if (Array.isArray(val) && val.length > largestLen) {
      largestKey = key;
      largestLen = val.length;
    }
  }

  if (!largestKey) return data;

  const arr = obj[largestKey] as unknown[];
  const result = { ...obj };
  const total = arr.length;

  // Binary search for max count that fits
  let lo = 1, hi = total;
  while (lo < hi) {
    const mid = Math.ceil((lo + hi) / 2);
    const candidate = { ...result, [largestKey]: arr.slice(0, mid) };
    if (JSON.stringify(candidate).length <= limitBytes) {
      lo = mid;
    } else {
      hi = mid - 1;
    }
  }

  result[largestKey] = arr.slice(0, lo);
  (result as any).truncatedAt = lo;
  if (!('totalNodeCount' in result)) {
    (result as any).totalNodeCount = total;
  }
  return result;
}
```

```typescript
// ─── Pagination cursor (v1) ──────────────────────────────────────────────────

export interface CursorData {
  offset: number;
}

/** Encode cursor data as base64 string with version prefix. */
export function encodeCursor(data: CursorData): string {
  const payload = `v1:${JSON.stringify(data)}`;
  return Buffer.from(payload, 'utf-8').toString('base64');
}

/** Decode a cursor string. Returns null if invalid or wrong version. */
export function decodeCursor(cursor: string): CursorData | null {
  try {
    const decoded = Buffer.from(cursor, 'base64').toString('utf-8');
    if (!decoded.startsWith('v1:')) return null;
    const data = JSON.parse(decoded.substring(3));
    if (typeof data.offset !== 'number') return null;
    return { offset: data.offset };
  } catch {
    return null;
  }
}
```

```typescript
// src/core/health-monitor.ts
/**
 * Health monitoring — Phase 4a
 *
 * Connection state machine: connected → degraded → reconnecting → disconnected
 * Statistics: request counts, sliding-window response times, error history
 * Heartbeat: periodic ping with automatic probe-interval fallback
 */

import { getLogger } from './logger.js';
import { isFeatureEnabled } from './feature-flags.js';

export type ConnectionState = 'disconnected' | 'connected' | 'degraded' | 'reconnecting';

export interface ErrorEntry {
  time: number;
  scope: string;
  type: string;
  message: string;
  retriable: boolean;
}

export interface HealthStats {
  totalRequests: number;
  successfulRequests: number;
  failedRequests: number;
  averageResponseTime: number;
  connectionStatus: ConnectionState;
  recentErrors: ErrorEntry[];
  uptime: number;
}

export interface HealthMonitorOptions {
  heartbeatIntervalMs?: number;   // default: 30000
  probeIntervalMs?: number;       // default: 60000
  maxConsecutiveFailures?: number; // default: 5
  degradedThreshold?: number;     // default: 3 (failures in last 10)
  sampleWindowSize?: number;      // default: 100
  errorHistorySize?: number;      // default: 20
}

export class HealthMonitor {
  private readonly heartbeatMs: number;
  private readonly probeMs: number;
  private readonly maxConsecFails: number;
  private readonly degradedThreshold: number;
  private readonly windowSize: number;
  private readonly errorSize: number;

  private state: ConnectionState = 'connected';
  private totalRequests = 0;
  private successfulRequests = 0;
  private failedRequests = 0;
  private responseTimes: number[] = [];
  private recentErrors: ErrorEntry[] = [];
  private consecutiveFails = 0;
  private startTime = Date.now();

  // For degraded detection: track last N request outcomes
  private recentOutcomes: boolean[] = []; // true = success, false = fail
  private static readonly OUTCOMES_WINDOW = 10;

  // Heartbeat
  private heartbeatTimer: ReturnType<typeof setInterval> | null = null;
  private baselineResponseTime: number | null = null;

  constructor(opts: HealthMonitorOptions = {}) {
    this.heartbeatMs = opts.heartbeatIntervalMs ?? 30_000;
    this.probeMs = opts.probeIntervalMs ?? 60_000;
    this.maxConsecFails = opts.maxConsecutiveFailures ?? 5;
    this.degradedThreshold = opts.degradedThreshold ?? 3;
    this.windowSize = opts.sampleWindowSize ?? 100;
    this.errorSize = opts.errorHistorySize ?? 20;
  }

  /** Record a successful request. */
  recordSuccess(responseTimeMs: number): void {
    this.totalRequests++;
    this.successfulRequests++;
    this.consecutiveFails = 0;
    this.pushResponseTime(responseTimeMs);
    this.pushOutcome(true);

    // Establish baseline from first 10 requests
    if (this.baselineResponseTime === null && this.successfulRequests >= 10) {
      this.baselineResponseTime = this.responseTimes.reduce((a, b) => a + b, 0) / this.responseTimes.length;
    }

    this.updateState();
  }

  /** Record a failed request. */
  recordFailure(errorType: string, message: string, scope = 'tool'): void {
    this.totalRequests++;
    this.failedRequests++;
    this.consecutiveFails++;
    this.pushOutcome(false);

    this.recentErrors.push({
      time: Date.now(),
      scope,
      type: errorType,
      message,
      retriable: true,
    });
    if (this.recentErrors.length > this.errorSize) {
      this.recentErrors.shift();
    }

    this.updateState();
  }

  /** Get current connection state. */
  getState(): ConnectionState {
    return this.state;
  }

  /** Manually set connection state. */
  setState(newState: ConnectionState): void {
    const old = this.state;
    this.state = newState;
    if (old !== newState) {
      getLogger().info('health', `State: ${old} → ${newState}`);
    }
  }

  /** Get health statistics snapshot. */
  getStats(): HealthStats {
    const avg = this.responseTimes.length > 0
      ? this.responseTimes.reduce((a, b) => a + b, 0) / this.responseTimes.length
      : 0;
    return {
      totalRequests: this.totalRequests,
      successfulRequests: this.successfulRequests,
      failedRequests: this.failedRequests,
      averageResponseTime: Math.round(avg),
      connectionStatus: this.state,
      recentErrors: [...this.recentErrors],
      uptime: Date.now() - this.startTime,
    };
  }

  /** Start periodic heartbeat. */
  startHeartbeat(pingFn: () => Promise<boolean>): void {
    if (!isFeatureEnabled('HEALTH_MONITOR')) return;
    this.stopHeartbeat();

    const tick = async () => {
      try {
        const ok = await pingFn();
        if (ok) {
          this.consecutiveFails = 0;
          if (this.state === 'reconnecting') this.setState('connected');
        } else {
          this.consecutiveFails++;
          this.recordFailure('HEARTBEAT_FAIL', 'Heartbeat ping failed', 'health');
        }
      } catch {
        this.consecutiveFails++;
        this.recordFailure('HEARTBEAT_ERROR', 'Heartbeat ping threw', 'health');
      }

      // Reschedule with appropriate interval
      const interval = this.state === 'reconnecting' ? this.probeMs : this.heartbeatMs;
      this.heartbeatTimer = setTimeout(tick, interval);
    };

    this.heartbeatTimer = setTimeout(tick, this.heartbeatMs);
  }

  /** Stop heartbeat. */
  stopHeartbeat(): void {
    if (this.heartbeatTimer) {
      clearTimeout(this.heartbeatTimer);
      this.heartbeatTimer = null;
    }
  }

  // ─── Internal ──────────────────────────────────────────────────────────────

  private pushResponseTime(ms: number): void {
    this.responseTimes.push(ms);
    if (this.responseTimes.length > this.windowSize) {
      this.responseTimes.shift();
    }
  }

  private pushOutcome(success: boolean): void {
    this.recentOutcomes.push(success);
    if (this.recentOutcomes.length > HealthMonitor.OUTCOMES_WINDOW) {
      this.recentOutcomes.shift();
    }
  }

  private updateState(): void {
    if (this.state === 'disconnected') return; // Manual recovery only

    // Check for reconnecting
    if (this.consecutiveFails >= this.maxConsecFails) {
      this.setState('reconnecting');
      return;
    }

    // Check for degraded
    const recentFails = this.recentOutcomes.filter(x => !x).length;

      ? this.responseTimes.reduce((a, b) => a + b, 0) / this.responseTimes.length
      : 0;

    if (this.state === 'connected') {
      if (recentFails >= this.degradedThreshold || slowResponse) {
        this.setState('degraded');
      }

      // Recover if recentFails < 2 and not slow
      const recovered = recentFails < 2 && (!this.baselineResponseTime || avgTime < 1.5 * this.baselineResponseTime);
      if (recovered) {
        this.setState('connected');
      }
```

```typescript
// src/core/reconnection-manager.ts
/**
 * Reconnection manager — Phase 4c
 *
 * Exponential backoff with jitter, max retries, and cancellation.
 * Delays: base → base*2 → base*4 → ... → maxDelay cap
 */

import { getLogger } from './logger.js';

export interface ReconnectionOptions {
  maxRetries?: number;    // default: 10
  baseDelayMs?: number;   // default: 800
  maxDelayMs?: number;    // default: 30000
}

export class ReconnectionManager {
  private readonly maxRetries: number;
  private readonly baseDelay: number;
  private readonly maxDelay: number;
  private attempt = 0;
  private cancelled = false;
  private currentTimer: ReturnType<typeof setTimeout> | null = null;

  constructor(opts: ReconnectionOptions = {}) {
    this.maxRetries = opts.maxRetries ?? 10;
    this.baseDelay = opts.baseDelayMs ?? 800;
    this.maxDelay = opts.maxDelayMs ?? 30_000;
  }

  /**
   * Start reconnection attempts.
   * @param connectFn Returns true if connection succeeded, false if failed.
   * @param onExhausted Called when all retries are exhausted.
   */
  start(connectFn: () => Promise<boolean>, onExhausted: () => void): void {
    this.attempt = 0;
    this.cancelled = false;
    this.scheduleNext(connectFn, onExhausted);
  }

  /** Cancel all pending reconnection attempts. */
  cancel(): void {
    this.cancelled = true;
    if (this.currentTimer) {
      clearTimeout(this.currentTimer);
      this.currentTimer = null;
    }
  }

  private scheduleNext(connectFn: () => Promise<boolean>, onExhausted: () => void): void {
    if (this.cancelled) return;

    const delay = Math.min(this.baseDelay * Math.pow(2, this.attempt), this.maxDelay);
    this.currentTimer = setTimeout(async () => {
      if (this.cancelled) return;

      this.attempt++;
      getLogger().info('reconnect', `Attempt ${this.attempt}/${this.maxRetries}`);

      try {
        const ok = await connectFn();
        if (this.cancelled) return;

        if (ok) {
          getLogger().info('reconnect', `Reconnected on attempt ${this.attempt}`);
          return; // success, stop retrying
        }
      } catch (err) {
        getLogger().warn('reconnect', `Attempt ${this.attempt} threw: ${err}`);
      }

      if (this.attempt >= this.maxRetries) {
        getLogger().error('reconnect', 'All reconnection attempts exhausted');
        onExhausted();
        return;
      }

      this.scheduleNext(connectFn, onExhausted);
    }, delay);
  }
}
```

```typescript
// src/core/middleware.ts
/**
 * Middleware pipeline executor — Phase 4
 *
 * Pipeline + post-hook model (not onion):
 * - before: sequential, first rejection stops pipeline
 * - after: all run even on rejection, can modify result
 */

import type { Middleware, DispatchContext, MiddlewareResult, ToolResult } from '../types.js';

/**
 * Execute middleware pipeline around a tool call.
 *
 * 1. Run each middleware.before() in order. Stop on first rejection.
 * 2. If all pass, execute the tool.
 * 3. Run ALL middleware.after() hooks (even if before rejected).
 *
 * @param middleware Ordered list of middleware
 * @param ctx Dispatch context
 * @param executeTool The actual tool execution function
 */
export async function executeMiddleware(
  middleware: Middleware[],
  ctx: DispatchContext,
  executeTool: () => Promise<ToolResult>,
): Promise<ToolResult> {
  let result: ToolResult | null = null;
  let rejected = false;

  // Before phase
  for (const mw of middleware) {
    const beforeResult: MiddlewareResult = await mw.before(ctx);
    if ('rejected' in beforeResult && beforeResult.rejected) {
      result = beforeResult.error;
      rejected = true;
      break;
    }
  }

  // Execute tool if not rejected
  if (!rejected) {
    result = await executeTool();
  }

  // After phase — always run all
  for (const mw of middleware) {
    if (mw.after) {
      try {
        result = await mw.after(ctx, result!);
      } catch {
        // After hooks must not crash the pipeline
      }
    }
  }

  return result!;
}

// Re-export types for convenience
export type { Middleware, DispatchContext };
```

```typescript
/**
 * MCP Prompt templates — Phase 5b
 *
 * 4 preset prompts for common Godot development tasks.
 * Returns structured guidance text (not code) for LLM execution.
 */

import type { PromptMessage } from '@modelcontextprotocol/sdk/types.js';

export interface PromptDef {
  name: string;
  description: string;
  arguments?: Array<{
    name: string;
    description: string;
    required?: boolean;
  }>;
}

const PROMPTS: Record<string, {
  def: PromptDef;
  build: (args: Record<string, string>) => PromptMessage[];
}> = {
  create_platformer: {
    def: {
      name: 'create_platformer',
      description: '2D platformer game scaffold guidance',
      arguments: [
        { name: 'project_name', description: 'Project name', required: false },
        { name: 'resolution', description: 'Target resolution (e.g. 1920x1080)', required: false },
      ],
    },
    build: (args) => [{
      role: 'user',
      content: { type: 'text', text: buildPlatformerGuide(args.project_name || 'platformer', args.resolution || '1280x720') },
    }],
  },

  setup_player_controller: {
    def: {
      name: 'setup_player_controller',
      description: 'Player controller setup guidance',
      arguments: [
        { name: 'dimension', description: '2d or 3d', required: false },
        { name: 'movement_type', description: 'topdown, platformer, or fps', required: false },
      ],
    },
    build: (args) => [{
      role: 'user',
      content: { type: 'text', text: buildPlayerControllerGuide(args.dimension || '2d', args.movement_type || 'platformer') },
    }],
  },

  optimize_scene: {
    def: {
      name: 'optimize_scene',
      description: 'Scene optimization analysis guidance',
      arguments: [
        { name: 'scene_path', description: 'Path to scene file (e.g. res://scenes/level1.tscn)', required: false },
      ],

    build: (args) => [{
      role: 'user',
      content: { type: 'text', text: buildOptimizeGuide(args.scene_path || 'res://scenes/main.tscn') },
    }],
  },

  debug_performance: {
    def: {
      name: 'debug_performance',
      description: 'Performance debugging walkthrough',
      arguments: [],
    },
    build: () => [{
      role: 'user',
      content: { type: 'text', text: buildDebugGuide() },
    }],
  },

// ─── Public API ──────────────────────────────────────────────────────────────

export function listPrompts(): PromptDef[] {
  return Object.values(PROMPTS).map(p => p.def);
}

export async function getPrompt(name: string, args: Record<string, string>): Promise<{ messages: PromptMessage[] }> {
  const prompt = PROMPTS[name];
  if (!prompt) throw new Error(`Unknown prompt: ${name}`);
  return { messages: prompt.build(args) };
}
```

```typescript
import type { Tool } from '@modelcontextprotocol/sdk/types.js';

/**
 * Create an elicitation middleware.
 * Checks required params vs provided args. If missing and client supports
 * elicitation, asks the client. Otherwise returns MISSING_PARAM error.
 *
 * Only prompts for primitive types (string/number/boolean/enum).
 * Skips object/array types.
 */
export function createElicitationMiddleware(
  getToolDef: (name: string) => Tool | null,
  elicitFn: ((params: string[]) => Promise<Record<string, string> | null>) | null,
): Middleware {
  return {
    name: 'elicitation',
    before: async (ctx) => {
      if (!isFeatureEnabled('ELICITATION')) return { passed: true };

      const def = getToolDef(ctx.toolName);
      if (!def?.inputSchema) return { passed: true };

      const schema = def.inputSchema as any;
      const required: string[] = schema.required ?? [];
      if (required.length === 0) return { passed: true };

      // Check which required params are missing
      const missing = required.filter(name => {
        const val = ctx.args[name];
        return val === undefined || val === null || val === '';
      });

      if (missing.length === 0) return { passed: true };

      // Filter to only primitive types
      const props = schema.properties ?? {};
      const primitiveMissing = missing.filter(name => {
        const prop = props[name];
        if (!prop) return false;
        const type = prop.type;
        return type === 'string' || type === 'number' || type === 'boolean' ||
          (type === 'string' && prop.enum);
      });

      if (primitiveMissing.length === 0) return { passed: true };

      // Try elicitation if client supports it
      if (elicitFn) {
        const elicited = await elicitFn(primitiveMissing);
        if (elicited) {
          // Merge elicited params (don't overwrite existing)
          for (const [key, val] of Object.entries(elicited)) {
            if (!(key in ctx.args)) {
              ctx.args[key] = val;
            }
          }
          return { passed: true };
        }
      }

      // No elicitation support or client declined — return error
      return {
        rejected: true,
        error: {
          content: [{ type: 'text', text: JSON.stringify({
            success: false,
            error: `Missing required parameter(s): ${primitiveMissing.join(', ')}`,
            error_code: 'MISSING_PARAM',
            missing_params: primitiveMissing,
          }) }],
        },
      };
    },
  };
}
```
